#include<cstdio>
#include<iostream>
#include<cstring>
#include<cstdlib>
#include<cmath>
#include<algorithm>
#define rep(i,a,b) for(LL i=a;i<=b;i++)
#define dep(i,a,b) for(LL i=a;i>=b;i--)
#define LL long long
#define mem(x,num) memset(x,num,sizeof x)
using namespace std;
const LL mod=998244353;
LL n,m;
LL phi(LL n)
{
	LL t=n;
	rep(i,2,sqrt(n))
	    if(n%i==0){
			t=t/i*(i-1);
			while(n%i==0)n/=i;
		}
	if(n!=1)t=t/n*(n-1);
	return t;
}

int main()
{
	cin>>n>>m;
	cout<<(phi(n)%mod)*(phi(m)%mod)%mod*(n%mod)%mod*(m%mod)%mod<<endl;
	return 0;
}
